<script>
    let currentBalance = 0;
    let timeLeft = 15;
    let adTimer = null;
    let integrityCheck = null;
    let adWindow = null;
    let isPaused = false;

    const overlay = document.getElementById('overlay');
    const timerText = document.getElementById('timerText');
    const statusWarning = document.getElementById('statusWarning');
    const adBtn = document.getElementById('adBtn');

    function startRewardedAd() {
        // Reset parameters
        timeLeft = 15;
        isPaused = false;
        statusWarning.innerText = "";
        adBtn.disabled = true;

        // 1. Open the Adsterra Direct Link/Popup
        // Replace with your real Adsterra Direct Link URL
        adWindow = window.open('https://www.google.com', '_blank', 'width=800,height=600');

        if (!adWindow) {
            alert("Popup Blocker detected! Please allow popups to earn rewards.");
            adBtn.disabled = false;
            return;
        }

        // 2. Open our strict internal UI overlay
        overlay.style.display = 'flex';
        timerText.innerText = `Watching Ad: ${timeLeft}s remaining`;

        // 3. Main Countdown Timer Loop
        adTimer = setInterval(() => {
            if (!isPaused) {
                timeLeft--;
                timerText.innerText = `Watching Ad: ${timeLeft}s remaining`;

                if (timeLeft <= 0) {
                    clearAllTimers();
                    grantReward();
                }
            }
        }, 1000);

        // 4. Integrity Checker: Runs every 400ms to catch manual closures
        integrityCheck = setInterval(() => {
            if (adWindow && adWindow.closed) {
                clearAllTimers();
                overlay.style.display = 'none';
                adBtn.disabled = false;
                alert("Exploit detected: You closed the advertisement window early. No rewards issued.");
            }
        }, 400);
    }

    // Tab Visibility Handler: Pauses counting instantly if user leaves the tab
    document.addEventListener("visibilitychange", () => {
        if (document.hidden && adTimer) {
            isPaused = true;
            statusWarning.innerText = "⚠️ TIMER PAUSED: You left the page! Return here to resume.";
        } else if (!document.hidden && adTimer) {
            isPaused = false;
            statusWarning.innerText = "";
        }
    });

    function clearAllTimers() {
        clearInterval(adTimer);
        clearInterval(integrityCheck);
        adTimer = null;
        integrityCheck = null;
    }

    function grantReward() {
        overlay.style.display = 'none';
        adBtn.disabled = false;
        
        currentBalance += 50;
        document.getElementById('balance').innerText = currentBalance;
        
        alert("Success! 50 Gems have been safely credited to your account.");
        
        // Context Note: In production, fire a secure network request here to log 
        // timestamps against your server/database to verify the full 15 seconds passed.
    }
</script>
